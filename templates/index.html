typescript 
// Initialize the NullByte client
import { NullByte } from '@nullbyte/sdk'

const nb = new NullByte({
  apiKey: process.env.NB_API_KEY,
  region: 'auto',
  timeout: 1000,
})

// Deploy a function to the edge
const fn = await nb.deploy({
  name: 'process-webhook',
  runtime: 'node20',
  handler: async (req, ctx) => {
    const body = await req.json()
    await ctx.store.put(body.id, body)
    return { ok: true, ts: Date.now() }
  },
})

console.log(fn.url)
